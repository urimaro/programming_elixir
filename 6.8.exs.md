同一モジュール内であれば、モジュール名を指定せずに関数を呼び出せる

```
> defmodule Mod do
> def func1 do
> IO.puts "in func1"
> end
> def func2 do
> func1()
> IO.puts "in func2"
> end
> end
{:module, Mod,
 <<70, 79, 82, 49, 0, 0, 4, 168, 66, 69, 65, 77, 65, 116, 85, 56, 0, 0, 0, 144,
   0, 0, 0, 16, 10, 69, 108, 105, 120, 105, 114, 46, 77, 111, 100, 8, 95, 95,
   105, 110, 102, 111, 95, 95, 9, 102, 117, ...>>, {:func2, 0}}
> Mod.func1
in func1
:ok
> Mod.func2
in func1
in func2
:ok
```
